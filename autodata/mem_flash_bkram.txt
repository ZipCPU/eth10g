@PREFIX=mem_flash_bkram
@MASTER.TYPE=SCRIPT
@MASTER.BUS=wbwide
@LD.FILE=nosdram.ld
@LD.DEFNS=
/* LD.DEFNS from zipmaster.txt */
_kram  = 0; /* No high-speed kernel RAM */
_ram   = ORIGIN(bkram);
_rom   = ORIGIN(flash);
_top_of_stack = ORIGIN(bkram) + LENGTH(bkram);

@LD.SCRIPT=
/* LD.SCRIPT from zipmaster.txt */
SECTIONS
{
       .rocode @$[0x%08x](zip.RESET_ADDRESS) : ALIGN(4) {
               _boot_address = .;
               *(.start) *(.boot)
               *(.text.startup)
               *(.text*)
               *(.rodata*) *(.strings)
       } > flash
       _kram_start = . ;
       _kram_end = . ;
       _ram_image_start = . ;
       .kernel : ALIGN_WITH_INPUT {
               *(.kernel)
               *(.data) *(COMMON)
               }> bkram AT> flash
       _ram_image_end = . ;
       .bss : ALIGN_WITH_INPUT {
               *(.bss)
               _bss_image_end = . ;
               } > bkram
       _top_of_heap = .;
}
